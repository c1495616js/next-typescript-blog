Validate if Personal Health Number valid at British Columbia.

Credit to my coworker, `Dr. Will`.

```ts
import _ from 'lodash';

const isValidPhn = (phn: string) =>
  !!phn.match(/^9[0-9]{9}$/) &&
  _(phn)
    .toArray()
    .drop(1)
    .map((d) => parseInt(d))
    .zipWith([2, 4, 8, 5, 10, 9, 7, 3, 1], (x, y) => x * y)
    .sum() %
    11 ===
    0;
```
