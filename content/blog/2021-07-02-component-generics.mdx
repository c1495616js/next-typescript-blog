```typescript
interface IProps<T = any> {
  data: T[];
}

export const BaseTable: FC<IProps<string>> = ({ data }) => {
  return <div></div>;
};

export const BaseTableWithGeneric = <T>({ data }: IProps<T>): ReactElement => {
  return <div></div>;
};

const MyComponent: FC = () => {
  return (
    <>
      <BaseTableWithGeneric<string> data={['string1', 'string2']} />

      <BaseTableWithGeneric<{ name: string }>
        data={[{ name: 'string1' }, { name: 'string2' }]}
      />
    </>
  );
};
```
